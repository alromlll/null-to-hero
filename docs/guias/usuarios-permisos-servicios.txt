# 2. Permisos: chmod, chown, chgrp
## Crea un playground para practicar sin romper nada:
mkdir ~/permisos-lab
cd ~/permisos-lab
touch archivo1.txt archivo2.sh secreto.txt

## Practica esto y entiende qué hace cada número:
chmod 755 archivo2.sh   # rwxr-xr-x → ejecutable por todos, editable solo por dueño
chmod 644 archivo1.txt  # rw-r--r-- → típico para archivos de config
chmod 600 secreto.txt   # rw------- → solo el dueño puede leerlo (claves SSH, etc.)

## Cambia el dueño
sudo chown two archivo1.txt
sudo chown two:sudo archivo2.sh  # dueño:grupo

## Comprueba con:
ls -la

## El truco para memorizar el octal: 4=leer, 2=escribir, 1=ejecutar. Los sumas para cada grupo (dueño/grupo/otros). 7=4+2+1=todo, 5=4+1=leer+ejecutar, 4=solo leer.

# 2.1 Permisos Linux en profundidad
## Además del octal básico, hay tres permisos especiales que debes conocer conceptualmente:

- SUID (4000): un ejecutable corre con los permisos del dueño, no del que lo ejecuta. Ejemplo: /usr/bin/passwd — cualquier usuario puede cambiar su contraseña aunque el archivo de contraseñas sea de root.
- SGID (2000): similar pero con el grupo. Útil en directorios compartidos.
- Sticky bit (1000): en un directorio, solo el dueño de un archivo puede borrarlo aunque otros tengan permiso de escritura. Lo verás en /tmp.

ls -la /tmp        # verás la 't' al final de los permisos
ls -la /usr/bin/passwd  # verás una 's' donde iría la 'x' del dueño

## systemd — cómo arranca Linux
systemd es el proceso número 1 (PID 1) en casi todas las distros modernas. Gestiona el arranque y todos los servicios. Todo en systemd son units — archivos de configuración que describen un servicio, un punto de montaje, un timer, etc.

##Los servicios están en /lib/systemd/system/ o /etc/systemd/system/. Mira el de nginx:
cat /lib/systemd/system/nginx.service
##Verás secciones [Unit], [Service] y [Install] — eso es toda la definición del servicio.

# 4. Instalar nginx y practicar systemctl
sudo apt update && sudo apt install nginx -y

## El ciclo de vida de un servicio — memoriza estos:
sudo systemctl start nginx      # arranca
sudo systemctl stop nginx       # para
sudo systemctl restart nginx    # para y arranca
sudo systemctl reload nginx     # recarga config sin parar (cuando se puede)
sudo systemctl enable nginx     # que arranque solo al iniciar el sistema
sudo systemctl disable nginx    # que no arranque solo
sudo systemctl status nginx     # estado actual + últimas líneas de log

## Verifica que nginx está sirviendo algo:
curl http://localhost
